 34:             'ep_Search' => 'ep_Search.php',
 35:         );

 38:     /**
 39:      * @param string $dir directory path to generate class map from
 40:      * @return array:array list of 3 associative arrays containing all mappings, correct mappings and bad files respectively
 41:      */
 42:     private function getClassMap($dir) {
 43:         $flags = RecursiveDirectoryIterator::KEY_AS_FILENAME 
 44:             | RecursiveDirectoryIterator::SKIP_DOTS 
 45:             | RecursiveDirectoryIterator::UNIX_PATHS;
 46:         $files = new RecursiveDirectoryIterator($dir, $flags);
 47:         $files = new RecursiveIteratorIterator($files, RecursiveIteratorIterator::LEAVES_ONLY);
 48:         
 49:         //use only *.php files (eg. for filter .dot.files)
 50:         $files = new RegexIterator($files, '/^.+\.php$/i', RegexIterator::MATCH);
 51:         
 52:         //use class name (without .php extension) as key
 53:         $files = new RegexIterator($files, '/(\.php)$/i', RegexIterator::REPLACE, RegexIterator::USE_KEY);
 54:         $files->rewind();
 55:         
 56:         $map = array();
 57:         $map_correct = array();
 58:         $regExp = '/class\s+([a-zA-Z0-9_]*)\s/i';
 59:         
 60:         //iterate over files
 61:         foreach ($files as $key => $fileInfo) {
 62:             $map[$key] = $files->getSubPathName();
 63:                 
 64:             $file = new SplFileObject($fileInfo->getRealPath());
 65:             $m = new RegexIterator($file, $regExp, RegexIterator::MATCH);
 66:             $class_matches = iterator_to_array($m, false);
 67: 
 68:             if (empty($class_matches)) {
 69:                 trigger_error("No class found in file ". $fileInfo->getRealPath(), E_USER_WARNING);
 70:             } else {
 71:                 $map_correct[$key] = $map[$key];
 72:             }
 73:         }
 74:         //error's map
 75:         $map_invalid = array_diff($map, $map_correct);
 76:         
 77:         return array($map, $map_correct, $map_invalid);
 78:     }
