 21:      * @var array Autoloader class map with class names as keys and paths to the files as values

 25:     public function __construct() {
 26:         $this->classMap = $this->getDefaultMap();

 36:     /**
 37:      * Default classes expected in eP_API. We map them manually since some of the classes 
 38:      * have different name and file name.
 39:      * @return array Autoloader class map with class names as keys and paths to the files as values 
 40:      */
 41:     protected function getDefaultMap() {
 42:         return array (
 43:             'ep_Api' => 'ep_Api.php',
 44:             'ep_Autoloader' => 'ep_Autoloader.php',
 45:             'ep_Dataset' => 'ep_Dataset.php',
 46:             'ep_Object' => 'ep_Object.php',
 47:             'ep_Search' => 'ep_Search.php',
 48:         );
 49:     }
 50:     
 51:     /**
 52:      * @param string $dir directory path to generate class map from
 53:      * @return array:array list of 3 associative arrays containing all mappings, correct mappings and bad files respectively
 54:      */
 55:     private function getClassMap($dir) {
 56:         $flags = RecursiveDirectoryIterator::KEY_AS_FILENAME 
 57:             | RecursiveDirectoryIterator::SKIP_DOTS 
 58:             | RecursiveDirectoryIterator::UNIX_PATHS;
 59:         $files = new RecursiveDirectoryIterator($dir, $flags);
 60:         $files = new RecursiveIteratorIterator($files, RecursiveIteratorIterator::LEAVES_ONLY);
 61:         
 62:         //use only *.php files (eg. for filter .dot.files)
 63:         $files = new RegexIterator($files, '/^.+\.php$/i', RegexIterator::MATCH);
 64:         
 65:         //use class name (without .php extension) as key
 66:         $files = new RegexIterator($files, '/(\.php)$/i', RegexIterator::REPLACE, RegexIterator::USE_KEY);
 67:         $files->rewind();
 68:         
 69:         $map = array();
 70:         $map_correct = array();
 71:         $regExp = '/class\s+([a-zA-Z0-9_]*)\s/i';
 72:         
 73:         //iterate over files
 74:         foreach ($files as $key => $fileInfo) {
 75:             $map[$key] = $files->getSubPathName();
 76:                 
 77:             $file = new SplFileObject($fileInfo->getRealPath());
 78:             $m = new RegexIterator($file, $regExp, RegexIterator::MATCH);
 79:             $class_matches = iterator_to_array($m, false);
 80: 
 81:             if (empty($class_matches)) {
 82:                 trigger_error("No class found in file ". $fileInfo->getRealPath(), E_USER_WARNING);
 83:             } else {
 84:                 $map_correct[$key] = $map[$key];
 85:             }
 86:         }
 87:         //error's map
 88:         $map_invalid = array_diff($map, $map_correct);
 89:         
 90:         return array($map, $map_correct, $map_invalid);
 91:     }

141:     /**
142:      * Autoload callback to be registered with spl_autoload_register
143:      * @param string $class
144:      * @return boolean
145:      */
146:     public function autoload($class) {
147:         if (isset($this->classMap[$class])) {
148:             if (!@include_once($this->classMap[$class])) {
149:                 return false;
150:             }
151:         } else {
152:             $path = dirname(__FILE__).'/objects/'.$class.'.php';
153:             if (!file_exists($path)) {
154:                 return false;
155:             }
156:             if (!@include_once($path)) {
157:                 return false;
158:             }
159:         }
160:     }
